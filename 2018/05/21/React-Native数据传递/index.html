React-Native数据传递

Posted on 2018-05-21 | Edited on 2019-08-29 | In 大前端 , React-Native

通过react-navigation传递数据

顺传

传递数据

_onItemPress = (item) => {
    NavigationService.navigate("Page1", {
        data: item
    });
};

获取数据

componentDidMount() {
    // 或者使用 this.props.navigation.getParam('data')
    let data = this.props.navigation.state.params.data;
}

逆传

传递数据

_callback = (message) => {
    console.log(message);
}

_onItemPress = (item) => {
    NavigationService.navigate("Page1", {
        callback: this._callback
    });
};

子组件获取回调并执行

render() {
    return (
        <View style={styles.container}>
            <Button
                title={'点击测试'}
                onPress={() => this.props.navigation.state.params.callback('hello world') }>
            </Button>
        </View>
    );
}

通过组件传递数据

顺传

传递数据

_renderItem = ({item, index}) => {
    return (<View>
        <Component1 msg={item} index={index} dataA={this.state.dataA} navigation={this.props.navigation}/>
    </View>)
};

子组件获取数据

export default class Component1 extends Component<Props> {
    constructor(props) {
        super(props);
        this.state = {
            msg: this.props.msg,
            index: this.props.index,
            dataA: this.props.dataA,
        }
    }

    //父组件更新数据，子组件不刷新时使用
    componentWillReceiveProps(nextProps) {
        this.setState({
            msg: nextProps.msg,
            index: nextProps.index,
            dataA: nextProps.dataA,
        });
    }
}

逆传

组件A
传递callback

_callback = (message) => {
    console.log(message);
}

render() {
    return (
    <View>
        <ComponentB callback={this._callback}></ComponentB>
    </View>)
};

组件B
子组件获取传递过来的callback并执行

class ComponentB extends React.PureComponent {
    _onPress = () => {
        this.props.callback('hello world')
    };

    render() {
        return (
            <View style={styles.container} >
                <Button
                    title={"点击测试"}
                    onPress={this._onPress}
                >
                </Button>
            </View>
        );
    }
}
